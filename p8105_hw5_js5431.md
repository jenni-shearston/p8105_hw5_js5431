``` r
file_names = list.files("./data")

paths = tibble(file_names) %>% 
  mutate(file_names = str_c("./data/", file_names))
```

``` r
readin = function(x) {
  read_csv(file = x)
}
```

``` r
study_data = 
  paths %>% 
  mutate(data = map(file_names, readin),
         id = str_replace(file_names, "./data/", ""),
         id = str_replace(id, ".csv", ""),
         exp_group = str_detect(id, "exp")) %>% 
  select(-file_names)
```
